76 uInt32 pciRead(int bus, int dev, int func, int reg, int bytes) {
77  uInt16 base;
78 
79  union {
80  struct confadd c;
81  uInt32 n;
82  } u;
83 
84  u.n = 0;
85  u.c.enable = 1;
86  u.c.rsvd = 0;
87  u.c.bus = bus;
88  u.c.dev = dev;
89  u.c.func = func;
90  u.c.reg = reg & 0xFC;
91 
92  outportDWord(0xCF8, u.n);
93 
94  base = 0xCFC + (reg & 0x03);
95 
96  switch (bytes) {
97  case 1:
98  return (inportByte(base));
99  case 2:
100  return (inportWord(base));
101  case 4:
102  return (inportDWord(base));
103  default:
104  return 0;
105  }
