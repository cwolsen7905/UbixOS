82  kprintf("eax: %08lx ebx: %08lx ecx: %08lx edx: %08lx\n", regs->tf_eax, regs->tf_ebx, regs->tf_ecx, regs->tf_edx);

85  kprintf("cr0: 0x%X, cr2: 0x%X, cr3: 0x%X, cr4: 0x%X\n", rcr0(), rcr2(), rcr3(), rcr4());

107  cr2 = rcr2();
108  kprintf("CR2: 0x%X(0x%X)[0x%X]", cr2, _current->tss.eip, _current->tss.ldt);
109  if (_current->id == 7)
110  while (1)
111  asm("nop");
112 
113  if ((frame->tf_eflags & PSL_I) == 0) {
114  if (SEL_GET_PL(frame->tf_cs) == SEL_PL_USER || (frame->tf_eflags & PSL_VM)) {
115  kpanic("INT OFF! USER");
116  die_if_kernel("TEST", frame, 0x100);
117  }
118  else {
119  kpanic("INT OFF! KERN[0x%X]", trap_code);
120  die_if_kernel("TEST", frame, 0x200);
121  }
122  }
123 
124  kprintf("trap_code: %i(0x%X), EIP: 0x%X, CR2: 0x%X\n", frame->tf_trapno, frame->tf_trapno, frame->tf_eip, cr2);
125  if (frame->tf_trapno == 0xc) {
126  vmm_pageFault(frame, cr2);
127  }
128  else {
129  kpanic("TRAPCODE");
130  die_if_kernel("trapCode", frame, frame->tf_trapno);
131  endTask(_current->id);

133  }
