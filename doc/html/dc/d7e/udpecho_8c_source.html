52 
53  while (1) {
54  buf = netconn_recv(conn);
55  addr = netbuf_fromaddr(buf);
56  port = netbuf_fromport(buf);
57  netconn_connect(conn, addr, port);
58  netconn_send(conn, buf);
59  netbuf_copy(buf, buffer, sizeof(buffer));
