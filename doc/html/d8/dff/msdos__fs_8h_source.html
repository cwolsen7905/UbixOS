35 struct msdos_sb_info {

37  unsigned char fats, fat_bits; /* number of FATs, FAT bits (12 or 16) */
38  unsigned short fat_start, fat_length; /* FAT start & length (sec.) */

43  gid_t fs_gid;

45  unsigned short fs_umask;

48  struct wait_queue *fat_wait;
49  int fat_lock;
50  int prev_free; /* previously returned free cluster number */
51  int free_clusters; /* -1 if undefined */
52 };
