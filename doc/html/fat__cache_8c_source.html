39 //-----------------------------------------------------------------------------

42 int fatfs_cache_init(struct fatfs *fs, FL_FILE *file)
43 {
44 #ifdef FAT_CLUSTER_CACHE_ENTRIES
45  int i;
46 
47  for (i=0;i<FAT_CLUSTER_CACHE_ENTRIES;i++)
48  {
49  file->cluster_cache_idx[i] = 0xFFFFFFFF; // Not used

51  }
52 #endif
53 
54  return 1;
55 }
