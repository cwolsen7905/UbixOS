32 char *strtok_r(char *s, const char *delim, char **last) {
33  char *spanp, *tok;
34  int c, sc;
35 
36  if (s == NULL && (s = *last) == NULL)
37  return (NULL);
38 
39  /*
40  * Skip (span) leading delimiters (s += strspn(s, delim), sort of).
41  */
42  cont: c = *s++;
43  for (spanp = (char *) delim; (sc = *spanp++) != 0;) {
44  if (c == sc)
45  goto cont;
46  }
47 
48  if (c == 0) { /* no non-delimiter characters */
49  *last = NULL;
50  return (NULL);
51  }
52  tok = s - 1;
53 
54  /*
55  * Scan token (scan for delimiters: s += strcspn(s, delim), sort of).
56  * Note that delim must have one NUL; we stop if we see that, too.
57  */
58  for (;;) {
59  c = *s++;
60  spanp = (char *) delim;
61  do {
62  if ((sc = *spanp++) == c) {
63  if (c == 0)
64  s = NULL;
65  else
66  s[-1] = '\0';
67  *last = s;
68  return (tok);
69  }
70  } while (sc != 0);
71  }
72  /* NOTREACHED */
73  return (NULL);
74 }
