146  _file = fl_fopen(file, "r");
147  else if ((fd->mode & 0x2) == 0x2) {

150  else

152  }
153  else
154  kprintf("Invalid Mode?");
155

157  //kprintf("ERROR[%s:%i]: Open file: [%s] failed\n", __FILE__, __LINE__, file);

163  fd->perms = 0x1;
