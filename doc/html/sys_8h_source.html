369 /*
370  * Macros to set/get and increase/decrease variables in a thread-safe way.
371  * Use these for accessing variable that are used from more than one thread.
372  */
373 
374 #ifndef SYS_ARCH_INC
375 #define SYS_ARCH_INC(var, val) do { \
376  SYS_ARCH_DECL_PROTECT(old_level); \
377  SYS_ARCH_PROTECT(old_level); \
378  var += val; \
379  SYS_ARCH_UNPROTECT(old_level); \
380  } while(0)
381 #endif /* SYS_ARCH_INC */
382 
383 #ifndef SYS_ARCH_DEC
384 #define SYS_ARCH_DEC(var, val) do { \
385  SYS_ARCH_DECL_PROTECT(old_level); \
386  SYS_ARCH_PROTECT(old_level); \
387  var -= val; \
388  SYS_ARCH_UNPROTECT(old_level); \
389  } while(0)
390 #endif /* SYS_ARCH_DEC */
391 
392 #ifndef SYS_ARCH_GET
