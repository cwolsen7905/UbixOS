104 int vfs_addMount( struct vfs_mountPoint *mp ) {
105 
106  /* If There Are No Existing Mounts Make It The First */
107  if ( systemVitals->mountPoints == 0x0 ) {
108  mp->prev = 0x0;
109  mp->next = 0x0;
110  systemVitals->mountPoints = mp;
111  }
112  else {
113  mp->next = systemVitals->mountPoints;
114  systemVitals->mountPoints->prev = mp;
115  mp->prev = 0x0;
116  systemVitals->mountPoints = mp;
117  }
118  /* Return */
119  return (0x0);
120 }
