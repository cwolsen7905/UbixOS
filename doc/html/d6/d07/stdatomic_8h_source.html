168 #if defined(_KERNEL)

174  __atomic_is_lock_free(sizeof(*(obj)), obj)

177  __atomic_is_lock_free(sizeof((obj)->__val), &(obj)->__val)

180  ((void)(obj), sizeof((obj)->__val) <= sizeof(void *))

309 __extension__ ({ \
310  __typeof__(object) __o = (object); \

316 #endif
